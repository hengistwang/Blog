1. C++ 实现string 类的赋值运算符
   #+begin_src cpp
// 这个是初级的写法，如果new失败了的话原来的string里的东西也没了
 CMyString& CMyString::operator = (const CMyString& str)
{
    if(this == &str)
        return *this;

    delete []m_pData;
    m_pData = nullptr;

    m_pData = new char[strlen(str.m_pData) + 1];
    strcpy(m_pData, str.m_pData);

    return *this;
}
//其实下面这个才是最常用的写法,用一个临时变量来存储str，然后交换两者，出去if后直接调用strTemp的析构函数，并且内存如果分配失败，也不会导致删除原来的数据；
CMyString& CMyString::operator = (const CMyString& str)
{
    if(this!=&str)
    {
        CMyString strTemp(str);
        swap(m_pData,strTemp.m_pData);
    }
    return *this;
}

// 下面这个是线程安全的例子
#include <iostream>
#include <thread>
#include <mutex>
class Singleton
{

private:
    static Singleton * singleton_;
    static std::mutex mutex_;

protected:
    Singleton(const std::string value): value_(value) {}
    ~Singleton() {}
    std::string value_;

public:
    Singleton(Singleton &other) = delete;
    void operator=(const Singleton &) = delete;
    static Singleton *GetInstance(const std::string& value);
    void SomeBusinessLogic() {}

    std::string value() const{return value_;
    }
};

Singleton* Singleton::singleton_{nullptr};
std::mutex Singleton::mutex_;

Singleton *Singleton::GetInstance(const std::string& value)
{
    std::lock_guard<std::mutex> lock(mutex_);
    if (singleton_ == nullptr)
    {
        singleton_ = new Singleton(value);
    }
    return singleton_;
}

void ThreadFoo(){
    std::this_thread::sleep_for(std::chrono::milliseconds(1000));
    Singleton* singleton = Singleton::GetInstance("FOO");
    std::cout << singleton->value() << "\n";
}

void ThreadBar(){
    std::this_thread::sleep_for(std::chrono::milliseconds(1000));
    Singleton* singleton = Singleton::GetInstance("BAR");
    std::cout << singleton->value() << "\n";
}

int main()
{
    std::cout <<"If you see the same value, then singleton was reused (yay!\n" <<
                "If you see different values, then 2 singletons were created (booo!!)\n\n" <<
                "RESULT:\n";
    std::thread t1(ThreadFoo);
    std::thread t2(ThreadBar);
    t1.join();
    t2.join();

    return 0;
}

3. 数组中重复的数字
   一个长度为n的数组，并且数字都在(0,n-1)范围内，找出任意一个重复的数字.
   #+begin_src cpp
//首先想到的呢便是用一个set来找重复,便有了下面的代码:
class Solution {
public:
    int findRepeatNumber(vector<int>& nums) {
        unordered_set<int> m;
        for(auto i : nums) {
            if(m.emplace(i).second == false) return i;
        }
        return -1;
    }
};
//但是题目说了每个数字的范围，那么理想情况便是每个数字的位置下标应和数字大小是一样的，那只需要看是否每个数字都在自己应该再的位置就行了
class Solution {
public:
    int findRepeatNumber(vector<int>& nums) {
        int n = nums.size();
        for(int i = 0; i < n; ++i){
            while(nums[i] != i) {//之前这里用了个if，出了个bug，应该要一直swap直到该位置是该放那里的数字
                if(nums[i]!=nums[nums[i]]) swap(nums[i],nums[nums[i]]);
                else return nums[i];
            }
        }
        return -1;
    }
};
